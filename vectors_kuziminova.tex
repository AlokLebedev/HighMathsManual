\section*{Задания на вычисление скалярного произведения векторов}

\textbf{a) Даны векторы:}
\[
\bar{a} = (-3; 1), \quad \bar{b} = (4; -2)
\]
\textbf{}\\
\textbf{Найдем их скалярное произведение ((x1 × x2) + (y1 × y2)):}
\[
\bar{a} \cdot \bar{b} = (-3) \cdot 4 + 1 \cdot (-2) = -12 - 2 = -10
\]
\textbf{}\\
\textbf{b) Даны векторы:}
\[
\bar{m} = (-1; 1), \quad \bar{n} = (1; -1)
\]
\textbf{}\\
\textbf{Найдем их скалярное произведение ((x1 × x2) + (y1 × y2)):}
\[
\bar{m} \cdot \bar{n} = (-1) \cdot 1 + 1 \cdot (-1) = -1 - 1 = -2
\]
